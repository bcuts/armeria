17  package com.linecorp.armeria.server.docs;
18  
19  /**
20   * The type of a field or a return value.
21   */
22  public enum Type {
23      /**
24       * The 'void' type. Used only as a return type.
25       */
26      VOID,
27  
28      /**
29       * The 'boolean' type.
30       */
31      BOOL,
32  
33      /**
34       * The signed 8-bit integer type.
35       */
36      I8,
37  
38      /**
39       * The double float type.
40       */
41      DOUBLE,
42  
43      /**
44       * The signed 16-bit integer type.
45       */
46      I16,
47  
48      /**
49       * The signed 32-bit integer type.
50       */
51      I32,
52  
53      /**
54       * The signed 64-bit integer type.
55       */
56      I64,
57  
58      /**
59       * The 'string' type.
60       */
61      STRING,
62  
63      /**
64       * The 'binary' type.
65       */
66      BINARY,
67  
68      /**
69       * The 'struct' type.
70       */
71      STRUCT,
72  
73      /**
74       * The 'map' type.
75       */
76      MAP,
77  
78      /**
79       * The 'set' type.
80       */
81      SET,
82  
83      /**
84       * The 'list' type.
85       */
86      LIST,
87  
88      /**
89       * The 'enum' type.
90       */
91      ENUM;
92  
93      private final TypeInfo typeInfo = new TypeInfo() {
94          @Override
95          public Type type() {
96              return Type.this;
97          }
98  
99          @Override
100         public String signature() {
101             return name();
102         }
103 
104         @Override
105         public String toString() {
106             return signature();
107         }
108     };
109 
110     /**
111      * Returns the {@link TypeInfo} of this base type.
112      *
113      * @throws UnsupportedOperationException if this type is not a base type but one of the following:
114      * <ul>
115      *   <li>{@link #STRUCT}</li>
116      *   <li>{@link #MAP}</li>
117      *   <li>{@link #SET}</li>
118      *   <li>{@link #LIST}</li>
119      *   <li>{@link #ENUM}</li>
120      * </ul>
121      */
122     public TypeInfo asTypeInfo() {
123         switch (this) {
124             case STRUCT:
125             case MAP:
126             case SET:
127             case LIST:
128             case ENUM:
129                 throw new UnsupportedOperationException();
130         }
131 
132         return typeInfo;
133     }
134 }
